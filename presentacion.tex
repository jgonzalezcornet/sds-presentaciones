\section{Introducción}
\begin{frame}{Sistema Real}
    Estudio de fenómenos colectivos presentes en la naturaleza.
    \begin{itemize}
        \item Bandadas de pájaros
        \item Enjambres de insectos
        \item Colonias de bacterias
    \end{itemize}
    \begin{figure}[htbp]
        \centering
%        \includegraphics[width=0.4\textwidth]{resources/bandada.jpg}
    \end{figure}
\end{frame}

\begin{frame}{Fundamentos}
    El modelo se basa principalmente en dos ecuaciones:
    \begin{equation*}
        x_i(t+1) = x_i(t) + v_i(t) \Delta t
    \end{equation*}
    \begin{equation*}
        \theta(t+1) = \langle \theta(t) \rangle_r+ \Delta \theta
    \end{equation*}
    La dirección promedio se obtiene:
    \begin{equation*}
        \langle \theta(t) \rangle_r = \arctan \left( \frac{\langle \sin \theta(t) \rangle_r}{\langle \cos \theta(t) \rangle_r} \right)
\end{equation*}
\end{frame}

\begin{frame}{Algoritmo Implementado}
    \begin{algorithm}[H]
    \small
        \caption{Paso de simulación — Modelo de Vicsek}
        \begin{algorithmic}[1]
            \State Calcular vecinos $\mathcal{N}_i$ de cada partícula $i$ con $r_c$
            \For{cada partícula $i \in \{1, \dots, N\}$}
                \State $\theta_i^{\text{new}} \gets \arctan\!\left(\dfrac{\displaystyle\sum_{j \in \mathcal{N}_i \cup \{i\}} \sin\theta_j}{\displaystyle\sum_{j \in \mathcal{N}_i \cup \{i\}} \cos\theta_j}\right) + \eta\left(U - \tfrac{1}{2}\right)$
            \EndFor
            \For{cada partícula $i \in \{1, \dots, N\}$}
                \State $\theta_i \gets \theta_i^{\text{new}}$
                \State $\vec{x}_i \gets \vec{x}_i + \vec{v}_i \,\Delta t \pmod{L}$
            \EndFor
        \end{algorithmic}
    \end{algorithm}
    \vspace{0.2cm}
    \footnotesize{$U \sim \mathcal{U}(0,1)$; condiciones de contorno periódicas}
\end{frame}

\section{Simulaciones}

\begin{frame}{Simulaciones}
    Parámetros de entrada fijos:
    \begin{itemize}
        \item $dt=1$ - Paso temporal
        \item $r_c=1$ - Radio de Interacción
        \item $v=0.03$ - Módulo de la Velocidad
        \item $L=10$ - Longitud del Lado de la Celda
    \end{itemize}
    Parámetros de entrada variables:
    \begin{itemize}
        \item $\rho\in[2, 8]$ - Densidad de Partículas
        \item $T\in[300, 2000]$ - Cantidad de Pasos
        \item $\eta\in[0.1, 5]$ - Ruido
        \item $N = \lfloor \rho \cdot L^2 \rfloor$ - Cantidad de Partículas
    \end{itemize}
\end{frame}

\begin{frame}{Observable Definido}
    Parámetro de Orden $v_a(t)$
    \begin{equation*}
        v_a = \frac{1}{Nv} \left|\sum_{i=1}^{N} \vec{v_i} \right|
    \end{equation*}
    \begin{itemize}
        \item Orden: Tiende a uno.
        \item Desorden: Tiende a cero.
    \end{itemize}
\end{frame}

\section{Conclusiones}
\begin{frame}{Conclusiones}
    \begin{itemize}
        \item El parámetro de polarización $v_a$ decrece con $\eta$: el ruido destruye el orden colectivo.
        \item El impacto del líder sobre la polarización global es limitado independientemente del ruido.
        \item Una mayor densidad $\rho$ favorece el orden colectivo.
    \end{itemize}
\end{frame}
